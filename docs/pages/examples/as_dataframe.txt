
.. code-block:: python
    :caption: TableData to pandas.DataFrame

    from pytablereader import TableData

    TableData(
        table_name="sample",
        header_list=["a", "b"],
        record_list=[[1, 2], [3.3, 4.4]]
    ).as_dataframe()


.. code-block:: none
    :caption: Output

         a    b
    0    1    2
    1  3.3  4.4
